\usepackage{luacode}
\begin{luacode*}
function sum(x)
    assert(x == math.floor(x), "x must be an integer")
    res = 0
    for k = 1, x do
        res = res + (-1)^(k+1)/k
    end
    tex.sprint(res)
end
\end{luacode*}
